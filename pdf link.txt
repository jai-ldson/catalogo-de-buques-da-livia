        function filterBouquets() {
            const searchTerm = searchInput.value.toLowerCase();
            const priceRange = priceFilter.value;
            
            let filtered = bouquets.filter(bouquet => {
                const matchesSearch = bouquet.name.toLowerCase().includes(searchTerm) || 
                                    bouquet.description.toLowerCase().includes(searchTerm);
                
                let matchesPrice = true;
                if (priceRange) {
                    if (priceRange === '0-100') {
                        matchesPrice = bouquet.price <= 100;
                    } else if (priceRange === '100-200') {
                        matchesPrice = bouquet.price > 100 && bouquet.price <= 200;
                    } else if (priceRange === '200+') {
                        matchesPrice = bouquet.price > 200;
                    }
                }
                
                return matchesSearch && matchesPrice;
            });
            
            renderFilteredBouquets(filtered);
        }

        // WhatsApp Helper Function - Universal Compatibility
        function openWhatsAppLink(url) {
            // Extract phone and message from URL
            const phoneMatch = url.match(/wa\.me\/([+\d]+)/);
            const phone = phoneMatch ? phoneMatch[1] : '+5566992818222';
            const urlParams = new URLSearchParams(url.split('?')[1] || '');
            const text = urlParams.get('text') || '';
            
            // Show confirmation message
            showWhatsAppMessage();
            
            // Multiple fallback strategies for maximum compatibility
            const strategies = [
                // Strategy 1: WhatsApp app protocol (most direct)
                () => {
                    const appUrl = `whatsapp://send?phone=${phone}&text=${encodeURIComponent(text)}`;
                    window.location.href = appUrl;
                },
                
                // Strategy 2: wa.me with window.open
                () => {
                    const waUrl = `https://wa.me/${phone}?text=${encodeURIComponent(text)}`;
                    window.open(waUrl, '_blank');
                },
                
                // Strategy 3: web.whatsapp.com
                () => {
                    const webUrl = `https://web.whatsapp.com/send?phone=${phone}&text=${encodeURIComponent(text)}`;
                    window.open(webUrl, '_blank');
                }
            ];
            
            // Try strategies with delays
            let strategyIndex = 0;
            
            function tryNextStrategy() {
                if (strategyIndex < strategies.length) {
                    try {
                        strategies[strategyIndex]();
                        strategyIndex++;
                        
                        // If still on page after 1.5 seconds, try next strategy
                        setTimeout(() => {
                            if (document.visibilityState === 'visible' && strategyIndex < strategies.length) {
                                tryNextStrategy();
                            } else if (document.visibilityState === 'visible' && strategyIndex >= strategies.length) {
                                // All strategies failed, show manual instructions
                                showManualInstructions(phone, text);
                            }
                        }, 1500);
                        
                    } catch (error) {
                        // Strategy failed, try next one immediately
                        strategyIndex++;
                        if (strategyIndex < strategies.length) {
                            setTimeout(tryNextStrategy, 100);
                        } else {
                            showManualInstructions(phone, text);
                        }
                    }
                } else {
                    // All strategies exhausted
                    showManualInstructions(phone, text);
                }
            }
            
            // Start trying strategies
            tryNextStrategy();
        }
